import Country from "@site/src/components/Country";
import SortableTable from "@site/src/components/SortableTable";
import services from "@site/src/data/self-driving-services.json";

# Self Driving Service

List of self-driving taxi services that are open to the public, waitlisting, or announced.

<SortableTable
  columns={[
    { header: "Service", accessorKey: "service" },
    { header: "Technology", accessorKey: "technology" },
    {
      header: "Area",
      accessorFn: (row) => [row.country, row.city],
      size: 200,
      cell: ({ getValue }) => {
        const [country, city] = getValue();
        return (
          <>
            <Country iso={country} name={false} /> {city}
          </>
        );
      },
    },
    {
      header: "Status",
      accessorKey: "status",
      cell: ({ getValue }) =>
        ({ public: "🌎", waitlist: "⏳", announced: "📢" })[getValue()],
    },
    { header: "Notes", accessorKey: "notes", size: 300 },
  ]}
  data={services}
  defaultSorting={{ id: "service", desc: false }}
/>
